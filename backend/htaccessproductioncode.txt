RewriteCond %{HTTP:Authorization} ^(._)
RewriteRule ^(._) - [E=HTTP_AUTHORIZATION:%1]


# Block direct access to PHP files from external users

<FilesMatch "\.php$">
Require all denied
</FilesMatch>
